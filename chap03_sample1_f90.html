 1program sample
 2  implicit none ! 暗黙の型宣言禁止
 3
 4  ! 変数を使う前には必ず以下のように宣言を行う
 5  integer :: n ! 整数型の変数の宣言
 6  real :: x    ! 実数型の変数の宣言
 7
 8  ! 代入
 9  n = 10
10  x = 3.14
11
12  ! 表示
13  write(*, *) 'integer => ', n
14  write(*, *) 'real => ', x
15
16  stop
17endprogram sample
